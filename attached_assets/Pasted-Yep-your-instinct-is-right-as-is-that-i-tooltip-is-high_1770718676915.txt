typescript
// Headline quality filter:
const isHeadlineRelevant = (headline: string, person: Person) => {
  const nameParts = person.name.toLowerCase().split(' ');
  const headlineLower = headline.toLowerCase();
  
  // Must contain full name or last name
  const hasName = nameParts.every(part => headlineLower.includes(part)) ||
                  headlineLower.includes(nameParts[nameParts.length - 1]);
  
  // Reject spam patterns
  const isSpam = /win|prize|contest|click here|free/i.test(headline);
  
  // Reject if too old (show only <24h headlines)
  const isRecent = headline.timestamp > Date.now() - 24 * 60 * 60 * 1000;
  
  return hasName && !isSpam && isRecent;
};

// In tooltip:
const relevantHeadlines = person.headlines
  .filter(h => isHeadlineRelevant(h, person))
  .slice(0, 2);

{relevantHeadlines.length > 0 ? (

        href={h.url} 
        target="_blank"
        className="block text-xs hover:underline mb-1"

      </a>
    ))}
  </div>

  </div>

Critical Backend Improvement (Regardless of Which Option)
Fix GDELT Queries to Reduce "Wrong Person" Results
Current (presumably):

typescript
const query = person.name; // "Logan Paul"
Problem: Returns results for any Logan, any Paul, and spam.

Better:

typescript
const query = "${person.name}" ${getDisambiguationTerms(person)};

function getDisambiguationTerms(person: Person) {
  const terms = [person.name]; // full name in quotes
  
  if (person.category === 'Sports') {
    terms.push(person.sport || 'athlete'); // "Logan Paul" boxing
  } else if (person.category === 'Entertainment') {
    terms.push('actor OR singer OR celebrity');
  } else if (person.category === 'Politics') {
    terms.push('politician OR government');
  }
  
  return terms.join(' ');
}
Example: "Logan Paul" boxing returns far fewer false positives than just Logan Paul.
