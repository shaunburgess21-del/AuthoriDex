typescript
// Compute thresholds from current visible data:
const sortedByRankChange = visiblePeople.sort((a,b) => Math.abs(b.rankChange) - Math.abs(a.rankChange));
const sortedByDelta = visiblePeople.sort((a,b) => Math.abs(b.change24h) - Math.abs(a.change24h));

// Top 10% get indicators:
const topRankMoverThreshold = Math.abs(sortedByRankChange[Math.floor(visiblePeople.length * 0.10)].rankChange);
const topDeltaThreshold = Math.abs(sortedByDelta[Math.floor(visiblePeople.length * 0.10)].change24h);

// Show indicator if in top 10% on either metric:
const showIndicator = Math.abs(rankChange) >= topRankMoverThreshold || 
                      Math.abs(change24h) >= topDeltaThreshold;
